Trick.grid.GridPanel Sample - Omega Trick for Ext JS

参照：Company.js,Branches.js

プラグインに、Trick.plugins.xcolumnsが初期状態で設定されています。

new Trick.grid.GridPanel({
    title: 'Trick.grid.GridPanel',
    width: 560,
    height: 150,
    store: {
        xtype: 'arraystore',
        fields: [
            'company',
            'branches'
        ],
        data: [
            ['Alcoa Inc', 1],
            ['Boeing Co.', 5],
            ['Hewlett-Packard Co.', 2],
            ['Wal-Mart Stores, Inc.', 9]
        ]
    },
    colModel: new Ext.grid.ColumnModel({
        xcolumns: [
            'companycolumn',
            'branchescolumn'
        ]
    }),
    renderTo: 'renderarea'
});
